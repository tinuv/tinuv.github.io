但是字符转整形硬是搞了30分钟
```cpp
#include <iostream>
#include <string>
#include <map>

using namespace std;

int main() {
    string origin;
    cin >> origin;
    int result = 0;
    for(int i = 0; i < origin.size(); i++) {
        char a[1];
        a[0] = origin[i];
        result = result + stoi(a);
    }
    string res = to_string(result);
    map<char, string> m = {{'0', "zero"}}, {'1', "one"}, {'2', "two"}, {'3', "three"}, {'4', "four"}, {'5', "five"}, {'6', "six"}, {'7', "seven"}, {'8', "eight"},{'9', "nine"}};
    for(int i = 0; i < res.size(); i++) {
        if(i != 0) {
            cout << " ";
        }
        cout << m[res[i]];
    }
    return 0;
}
```
